(Yeah)", ] def typing_animation(text, delay=0.08): for char in text: print(char, end='', flush=True) time.sleep(delay) print() def show_lyrics(lyrics): print("") for line in lyrics: typing_animation(line) time.sleep(1.5) print("") if __name__ == "__main__": show_lyrics(lyrics)
